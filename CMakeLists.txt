cmake_minimum_required(VERSION 2.8.9)
project(relz)
set(CMAKE_BUILD_TYPE Release)

include(CheckCXXCompilerFlag)
CHECK_CXX_COMPILER_FLAG("-std=c++11" COMPILER_SUPPORTS_CXX11)
CHECK_CXX_COMPILER_FLAG("-std=c++0x" COMPILER_SUPPORTS_CXX0X)
if(COMPILER_SUPPORTS_CXX11)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11 -Wall -O5")
elseif(COMPILER_SUPPORTS_CXX0X)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++0x -Wall -O5")
else()
    message(FATAL_ERROR "Compiler ${CMAKE_CXX_COMPILER} has no C++11 support.")
endif()

set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(RLZ_INCLUDE_DIRS ${PROJECT_SOURCE_DIR}/include/ ${PROJECT_BINARY_DIR}/include/)
include_directories(${RLZ_INCLUDE_DIRS})

include_directories(${PROJECT_SOURCE_DIR}/include)
file(GLOB SOURCES "${PROJECT_SOURCE_DIR}/src/*.cpp")

add_executable(compress "${PROJECT_SOURCE_DIR}/programs/compress.cpp")
add_executable(decompress "${PROJECT_SOURCE_DIR}/programs/decompress.cpp")
add_executable(build_reference "${PROJECT_SOURCE_DIR}/programs/build_reference.cpp")
add_executable(test_sdsl "${PROJECT_SOURCE_DIR}/programs/test_sdsl.cpp")
add_executable(build_relz_basic "${PROJECT_SOURCE_DIR}/programs/build_relz_basic.cpp")
add_executable(build_relz_reference "${PROJECT_SOURCE_DIR}/programs/build_relz_reference.cpp")
add_executable(build_relz_reference_simple "${PROJECT_SOURCE_DIR}/programs/build_relz_reference_simple.cpp")
add_executable(build_relz_hash "${PROJECT_SOURCE_DIR}/programs/build_relz_hash.cpp")
add_executable(relz_search_basic "${PROJECT_SOURCE_DIR}/programs/relz_search_basic.cpp")
add_executable(relz_search_reference "${PROJECT_SOURCE_DIR}/programs/relz_search_reference.cpp")
add_executable(relz_search_reference_simple "${PROJECT_SOURCE_DIR}/programs/relz_search_reference_simple.cpp")
add_executable(relz_search_hash "${PROJECT_SOURCE_DIR}/programs/relz_search_hash.cpp")
add_executable(relz_search_hash_simple "${PROJECT_SOURCE_DIR}/programs/relz_search_hash_simple.cpp")
add_executable(clean_text "${PROJECT_SOURCE_DIR}/programs/clean_text.cpp")
add_executable(generate_queries "${PROJECT_SOURCE_DIR}/programs/generate_queries.cpp")
add_executable(test_karp_rabin "${PROJECT_SOURCE_DIR}/programs/test_karp_rabin.cpp")
add_executable(test_compacted_text "${PROJECT_SOURCE_DIR}/programs/test_compacted_text.cpp")

add_library(relz SHARED ${SOURCES})
link_directories(${CMAKE_BINARY_DIR})

target_link_libraries(compress relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(decompress relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(build_reference relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(test_sdsl relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(build_relz_basic relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(build_relz_reference relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(build_relz_reference_simple relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(build_relz_hash relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(relz_search_basic relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(relz_search_reference relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(relz_search_reference_simple relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(relz_search_hash relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(relz_search_hash_simple relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(clean_text)
target_link_libraries(generate_queries)
target_link_libraries(test_karp_rabin relz pthread sdsl divsufsort divsufsort64)
target_link_libraries(test_compacted_text relz pthread sdsl divsufsort divsufsort64)

if (CMAKE_INSTALL_PREFIX_INITIALIZED_TO_DEFAULT)
    set (CMAKE_INSTALL_PREFIX "/usr/" CACHE PATH "default install path" FORCE )
endif()

# Making this a local library
install(TARGETS relz DESTINATION ${INSTALL_INCLUDE_DIR}/lib)
install(TARGETS compress DESTINATION ${INSTALL_INCLUDE_DIR}/bin)
install(DIRECTORY include DESTINATION ${INSTALL_INCLUDE_DIR})
